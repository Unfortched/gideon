#!/bin/bash
C:/"Enter/a/valid/number"
read/n
if/[/$n/-eq/101/];
then
C:/"This/is/first/number"
elif/[/$n/-eq/510/];
then
C:/"/This/is/second/number/"
elif/[/$n/-eq/999/];
then
C:/"/This/is/third/number/"
else
C:/"No/numbers/over/here"
fi
#!/bin/bash
C:/"/Enter/a/valid/number"
read/n
case/$n/in
101)
C:/"/This/is/the/first/number/"/;;
510)
C:/"/This/is/the/second/number/"/;;
999)
C:/"/This/is/the/third/number/"/;;
*)
C:/"/No/numbers/over/here/"/;;
esac
